add_executable(winvnc4 WIN32
  buildTime.cxx
  ControlPanel.cxx
  ManagedListener.cxx
  QueryConnectDialog.cxx
  STrayIcon.cxx
  VNCServerService.cxx
  VNCServerWin32.cxx

  RUNTIME DESTINATION ${CMAKE_INSTALL_FULL_BINDIR}
    COMPONENT Runtime
)
